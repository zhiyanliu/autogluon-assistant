```python
from datasets import load_dataset

queries = load_dataset("json", data_files="ft_data/test_queries.jsonl")["train"]
corpus = load_dataset("json", data_files="ft_data/corpus.jsonl")["train"]
qrels = load_dataset("json", data_files="ft_data/test_qrels.jsonl")["train"]

queries_text = queries["text"]
corpus_text = [text for sub in corpus["text"] for text in sub]

# Build qrels dict: {qid: {docid: relevance}}
qrels_dict = {}
for line in qrels:
    if line['qid'] not in qrels_dict:
        qrels_dict[line['qid']] = {}
    qrels_dict[line['qid']][line['docid']] = line['relevance']
```

## Search with FAISS

Encodes queries/corpus into embeddings, indexes with FAISS (flat inner product), and retrieves top-100 results in batches of 32:

```python
import faiss, numpy as np
from tqdm import tqdm

def search(model, queries_text, corpus_text):
    queries_embeddings = model.encode_queries(queries_text)
    corpus_embeddings = model.encode_corpus(corpus_text)
    
    dim = corpus_embeddings.shape[-1]
    index = faiss.index_factory(dim, 'Flat', faiss.METRIC_INNER_PRODUCT)
    corpus_embeddings = corpus_embeddings.astype(np.float32)
    index.train(corpus_embeddings)
    index.add(corpus_embeddings)

    all_scores, all_indices = [], []
    for i in tqdm(range(0, len(queries_embeddings), 32), desc="Searching"):
        j = min(i + 32, len(queries_embeddings))
        score, indice = index.search(queries_embeddings[i:j].astype(np.float32), k=100)
        all_scores.append(score)
        all_indices.append(indice)

    all_scores = np.concatenate(all_scores, axis=0)
    all_indices = np.concatenate(all_indices, axis=0)
    
    results = {}
    for idx, (scores, indices) in enumerate(zip(all_scores, all_indices)):
        results[queries["id"][idx]] = {
            corpus["id"][index]: float(score)
            for score, index in zip(scores, indices) if index != -1
        }
    return results
```

## Evaluation: Raw vs Fine-tuned

```python
from FlagEmbedding.abc.evaluation.utils import evaluate_metrics, evaluate_mrr
from FlagEmbedding import FlagModel

k_values = [10, 100]
raw_name = "BAAI/bge-large-en-v1.5"
finetuned_path = "test_encoder_only_base_bge-large-en-v1.5"

# Common model params
model_kwargs = dict(
    query_instruction_for_retrieval="Represent this sentence for searching relevant passages:",
    devices=[0],
    use_fp16=False
)

# Evaluate both models
for path in [raw_name, finetuned_path]:
    model = FlagModel(path, **model_kwargs)
    results = search(model, queries_text, corpus_text)
    eval_res = evaluate_metrics(qrels_dict, results, k_values)
    mrr = evaluate_mrr(qrels_dict, results, k_values)
    for res in eval_res:
        print(res)
    print(mrr)
```

**Key details:**
- Uses `faiss.METRIC_INNER_PRODUCT` with flat index for exact search
- Retrieves top-100 candidates per query, evaluates at k=10 and k=100
- `encode_queries` / `encode_corpus` handle query instruction prefixing automatically
- Fine-tuned model shows clear improvement across all metrics (NDCG, MRR)
