# Evaluate on MLDR - Condensed Tutorial

## Overview
MLDR is a Multilingual Long-Document Retrieval dataset covering 13 languages, built from Wikipedia, Wudao, and mC4. Questions are generated via GPT-3.5 from sampled paragraphs.

## 1. Load Dataset

```python
from datasets import load_dataset

lang = "en"
dataset = load_dataset('Shitao/MLDR', lang, trust_remote_code=True)
corpus = load_dataset('Shitao/MLDR', f"corpus-{lang}", trust_remote_code=True)['corpus']

# Dataset structure: query_id, query, positive_passages, negative_passages
# Corpus structure: docid (format: doc-<language>-<id>), text

corpus_ids = corpus['docid']
corpus_text = corpus['text']
queries_ids = dataset['dev']['query_id']
queries_text = dataset['dev']['query']
```

## 2. Evaluate from Scratch

### 2.1 Embedding
```python
from FlagEmbedding import FlagModel

model = FlagModel('BAAI/bge-base-en-v1.5')
queries_embeddings = model.encode_queries(queries_text)
corpus_embeddings = model.encode_corpus(corpus_text)
```

### 2.2 Indexing (Faiss)
```python
import faiss, numpy as np

dim = corpus_embeddings.shape[-1]  # 768 for bge-base-en-v1.5
index = faiss.index_factory(dim, 'Flat', faiss.METRIC_INNER_PRODUCT)
corpus_embeddings = corpus_embeddings.astype(np.float32)
index.train(corpus_embeddings)
index.add(corpus_embeddings)
```

### 2.3 Searching
```python
from tqdm import tqdm

all_scores, all_indices = [], []
for i in tqdm(range(0, len(queries_embeddings), 32), desc="Searching"):
    j = min(i + 32, len(queries_embeddings))
    score, indice = index.search(queries_embeddings[i:j].astype(np.float32), k=100)
    all_scores.append(score)
    all_indices.append(indice)

all_scores = np.concatenate(all_scores, axis=0)
all_indices = np.concatenate(all_indices, axis=0)

results = {}
for idx, (scores, indices) in enumerate(zip(all_scores, all_indices)):
    results[queries_ids[idx]] = {
        corpus_ids[index]: float(score)
        for score, index in zip(scores, indices) if index != -1
    }
```

### 2.4 Evaluating
```python
import pytrec_eval
from collections import defaultdict

# Build qrels dict
qrels_dict = {}
for data in dataset['dev']:
    qid = str(data["query_id"])
    qrels_dict.setdefault(qid, {})
    for doc in data["positive_passages"]:
        qrels_dict[qid][str(doc["docid"])] = 1
    for doc in data["negative_passages"]:
        qrels_dict[qid][str(doc["docid"])] = 0

# Evaluate with pytrec_eval
ndcg_string = "ndcg_cut." + ",".join([str(k) for k in [10, 100]])
recall_string = "recall." + ",".join([str(k) for k in [10, 100]])

evaluator = pytrec_eval.RelevanceEvaluator(qrels_dict, {ndcg_string, recall_string})
scores = evaluator.evaluate(results)

all_ndcgs, all_recalls = defaultdict(list), defaultdict(list)
for query_id in scores:
    for k in [10, 100]:
        all_ndcgs[f"NDCG@{k}"].append(scores[query_id][f"ndcg_cut_{k}"])
        all_recalls[f"Recall@{k}"].append(scores[query_id][f"recall_{k}"])

for k in [10, 100]:
    all_ndcgs[f"NDCG@{k}"] = round(sum(all_ndcgs[f"NDCG@{k}"]) / len(scores), 5)
    all_recalls[f"Recall@{k}"] = round(sum(all_recalls[f"Recall@{k}"]) / len(scores), 5)
```

## 3. Evaluate using FlagEmbedding Runner

```python
from transformers import HfArgumentParser
from FlagEmbedding.evaluation.mldr import MLDREvalArgs, MLDREvalModelArgs, MLDREvalRunner

# Key parameters: dataset_names, splits, search_top_k, k_values, eval_metrics, devices, embedder_batch_size
parser = HfArgumentParser((MLDREvalArgs, MLDREvalModelArgs))
eval_args, model_args = parser.parse_args_into_dataclasses()

runner = MLDREvalRunner(eval_args=eval_args, model_args=model_args)
runner.run()
```

**Key CLI arguments:** `--eval_name mldr`, `--dataset_names en`, `--splits dev`, `--search_top_k 1000`, `--k_values 10 100`, `--eval_metrics ndcg_at_10`, `--embedder_name_or_path BAAI/bge-base-en-v1.5`, `--devices cuda:0 cuda:1`, `--embedder_batch_size 1024`
